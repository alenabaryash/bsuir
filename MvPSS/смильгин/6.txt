kol_vo_zan_st table P2,1,1,5
kol_vo_zan_man table X$zan_man,1,1,13

initial X$kol_st,4
m_man matrix ,1,4
initial MX$m_man(1,1),3
initial MX$m_man(1,2),3
initial MX$m_man(1,3),3
initial MX$m_man(1,4),3

man1 EQU 1
man2 EQU 2
man3 EQU 3
man4 EQU 4
man1 storage 3
man2 storage 3
man3 storage 3
man4 storage 3

generate 8,1; поступление деталей

select MIN 1,1,X$kol_st,,S

queue P1
test NE MX$m_man(1,P1),0
priority 2
msavevalue m_man-,1,P1,1
seize P1
depart P1
advance 2; установка детали
	enter P1
release P1
	advance 20; нанесение краски
priority 1
queue P1
seize P1
msavevalue m_man+,1,P1,1
	leave P1
depart P1
advance 4; съем готовой детали
release P1
terminate

generate 1
count SF 2,1,4; число станков со всеми тремя занятыми манипуляторами
TABULATE kol_vo_zan_st
savevalue zan_man,(S1+S2+S3+S4); общее число занятых красящих манипуляторов
TABULATE kol_vo_zan_man
terminate

generate (2#8#60)
terminate 1
start 1
